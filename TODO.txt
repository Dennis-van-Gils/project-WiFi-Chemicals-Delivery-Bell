    static bool request_is_pending = false;
    
    static bool request_LED_white = false;
    static bool state_LED_white = false;
   
    static bool request_LED_blue = false;
    static bool state_LED_blue = false;
   
    // Check physical buttons
    switch_white.poll();
    if (switch_white.pushed()) {
      request_LED_white = !state_LED_white;
    }
    
    // Check for incoming serial commands used for debugging
    if (sc.available()) {
      strCmd = sc.getCmd();

      if (strcmp(strCmd, "w") == 0) {
        request_LED_white = !state_LED_white;
      }
    }

    // Did anything change?
    if (request_LED_white != state_LED_white) {
      digitalWrite(PIN_LED_BTN_WHITE, request_LED_white);
      digitalWrite(PIN_LED_ONBOARD_RED, !request_LED_white);
      Ser.println("WHITE\n-----");
      request_is_pending = true;
    }
    
    if (request_is_pending) {
      // Send out new states to the web server
      if (send_buttons(request_LED_white, request_LED_blue)) {
        // Success -> grant request
        state_LED_white = request_LED_white;
        state_LED_blue = request_LED_blue;
      }
      digitalWrite(PIN_LED_BTN_WHITE, state_LED_white);
      digitalWrite(PIN_LED_ONBOARD_RED, !state_LED_white);
      digitalWrite(PIN_LED_BTN_BLUE, state_LED_blue);
      digitalWrite(PIN_LED_ONBOARD_BLUE, !state_LED_blue);
      
      // Reset pending request
      request_LED_white = state_LED_white;
      request_LED_blue = state_LED_blue;
      request_is_pending = false;
    }
